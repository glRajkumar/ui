## Usage

### Basic
```tsx
import { DropdownCheckboxWrapper, DropdownRadioWrapper, DropdownWrapper } from "@/components/ui/dropdown"
import { Button } from "@/components/ui/button"

export function Basic() {
  return (
    <>
      <DropdownWrapper
        options={["Option 1", "Option 2", "Option 3"]}
        contentProps={{ align: "end" }}
      >
        <Button variant="outline">
          Options
        </Button>
      </DropdownWrapper>

      <DropdownCheckboxWrapper options={dropdownOptions}>
        <Button variant="outline">
          Checkbox
        </Button>
      </DropdownCheckboxWrapper>

      <DropdownRadioWrapper options={dropdownOptions}>
        <Button variant="outline">
          Radio
        </Button>
      </DropdownRadioWrapper>
    </>
  )
}
```

### Controlled
```tsx
import { useState } from "react"
import { DropdownCheckboxWrapper, DropdownRadioWrapper, DropdownWrapper } from "@/components/ui/dropdown"
import { Button } from "@/components/ui/button"

export function Controlled() {
  const [checked, setChecked] = useState<allowedPrimitiveT[]>([])
  const [val, setVal] = useState<allowedPrimitiveT>(true)

  return (
    <>
      <DropdownWrapper
        options={dropdownOptions}
        onSelect={v => console.log(v)}
      >
        <Button variant="outline">
          Options
        </Button>
      </DropdownWrapper>

      <DropdownCheckboxWrapper
        checked={checked}
        options={dropdownOptions}
        onCheckedChange={(value, isChecked) => {
          setChecked((prev) =>
            isChecked
              ? [...prev, value]
              : prev.filter((x) => x !== value)
          )
        }}
      >
        <Button variant="outline">
          Checkbox
        </Button>
      </DropdownCheckboxWrapper>

      <DropdownRadioWrapper
        value={val}
        options={dropdownOptions}
        onValueChange={setVal}
      >
        <Button variant="outline">
          Radio
        </Button>
      </DropdownRadioWrapper>
    </>
  )
}
```

### Nested Complex Data
```tsx
const dropdownOptions = [
  { label: "New File", value: "new", shortcut: "Ctrl+N" },
  "Save",
  12,
  { label: <><Banana /> Banana</>, value: "banana" },
  "---",
  {
    group: "Settings",
    options: [
      { label: "Appearance", value: "appearance" },
      22,
      true
    ],
  },

  {
    submenu: "More",
    options: [
      { label: <><Apple /> Apple</>, value: "apple" },
      {
        group: "Tools",
        options: [
          { label: "Formatter", value: "formatter" },
          false,
        ],
      },
    ],
  },
]

export function Complex() {
  return (
    <DropdownWrapper
      options={dropdownOptions}
    />
  )
}
```

#### `DropdownWrapper`
<AutoTypeTable 
name="DropdownWrapper" 
type={`export interface DropdownWrapper {
  /**
   * Trigger element for the dropdown. Wrapped with asChild, so pass a single React element
   * 
   * \`\`\`tsx
   * <DropdownWrapper options={options}>
   *   <Button>Open Menu</Button>
   * </DropdownWrapper>
   * \`\`\`
   */
  children: React.ReactNode;

  /**
   * Array of dropdown options - supports primitives, objects, groups, and submenus
   * 
   * Supports:
   * - Primitives: \`string | number | boolean\`
   * - Objects: \`{ label: React.ReactNode, value: allowedPrimitiveT, className?: string }\`
   * - Objects with extras: \`{ label, value, variant?: "default" | "destructive", shortcut?: string, disabled?: boolean }\`
   * - Groups: \`{ group: string, options: Array, className?: string }\`
   * - Submenus: \`{ submenu: string, options: Array, triggerCls?: string, contentCls?: string }\`
   * - Separators: Use \`"---"\` as a value
   * 
   * \`\`\`tsx
   * const options = [
   *   "Simple item",
   *   { label: "Edit", value: "edit", shortcut: "âŒ˜E" },
   *   { label: "Delete", value: "delete", variant: "destructive" },
   *   "---",
   *   { 
   *     group: "Actions", 
   *     options: ["Save", "Export"] 
   *   },
   *   {
   *     submenu: "Share",
   *     options: ["Email", "Link", "Social"]
   *   }
   * ]
   * \`\`\`
   */
  options: dropdownOptionsT;

  /**
   * Additional CSS classes for individual dropdown items
   * 
   * Note: Individual options can override this with their own \`className\`
   */
  itemCls?: string;

  /**
   * Additional CSS classes applied to option groups
   */
  groupCls?: string;

  /**
   * Props passed to the DropdownMenuContent component
   * 
   * \`\`\`tsx
   * contentProps={{ 
   *   align: "end", 
   *   className: "w-56" 
   * }}
   * \`\`\`
   */
  contentProps?: React.ComponentProps<typeof DropdownMenuContent>;

  /**
   * Callback fired when an option is selected
   * Receives the value of the selected option
   * 
   * \`\`\`tsx
   * onSelect={(value) => {
   *   console.log("Selected:", value);
   * }}
   * \`\`\`
   */

  /**
   * All other props from DropdownMenu component
   */
  ...DropdownMenuProps?: React.ComponentProps<typeof DropdownMenu>
}`}

type={`export interface DropdownCheckboxWrapper {
  /**
   * Trigger element for the dropdown (must accept ref)
   * Wrapped with asChild, so pass a single React element
   */
  children: React.ReactNode;

  /**
   * Array of checkbox options - supports primitives, objects, groups, and submenus
   * 
   * Supports:
   * - Primitives: \`string | number | boolean\`
   * - Objects: \`{ label: React.ReactNode, value: allowedPrimitiveT, disabled?: boolean }\`
   * - Groups: \`{ group: string, options: Array, className?: string }\`
   * - Submenus: \`{ submenu: string, options: Array, triggerCls?: string, contentCls?: string }\`
   * - Separators: Use \`"---"\` as a value
   * 
   * \`\`\`tsx
   * const options = [
   *   "Email",
   *   "SMS",
   *   { label: "Push", value: "push", disabled: true },
   *   "---",
   *   { 
   *     group: "Marketing", 
   *     options: ["Newsletter", "Updates"] 
   *   }
   * ]
   * \`\`\`
   */
  options: inputOptionsT;

  /**
   * Optional label displayed at the top of the dropdown
   * Automatically adds a separator after the label
   * 
   * \`\`\`tsx
   * label="Select Notifications"
   * \`\`\`
   */
  label?: string;

  /**
   * Additional CSS classes for individual checkbox items
   */
  itemCls?: string;

  /**
   * Additional CSS classes applied to option groups
   */
  groupCls?: string;

  /**
   * Props passed to the DropdownMenuContent component
   */
  contentProps?: React.ComponentProps<typeof DropdownMenuContent>;

  /**
   * Controlled state: array of checked values
   * 
   * If provided, component operates in controlled mode
   * If not provided, manages its own internal state
   * 
   * \`\`\`tsx
   * checked={["email", "sms"]}
   * \`\`\`
   */
  checked?: (string | number | boolean)[];

  /**
   * Callback fired when a checkbox state changes
   * 
   * @param value - The value of the toggled option
   * @param checked - New checked state (true/false)
   * 
   * \`\`\`tsx
   * onCheckedChange={(value, checked) => {
   *   if (checked) {
   *     setSelected([...selected, value])
   *   } else {
   *     setSelected(selected.filter(v => v !== value))
   *   }
   * }}
   * \`\`\`
   */
  onCheckedChange?: (value: string | number | boolean, checked: boolean) => void;

  /**
   * All other props from DropdownMenu component
   */
  ...DropdownMenuProps?: React.ComponentProps<typeof DropdownMenu>
}`}
