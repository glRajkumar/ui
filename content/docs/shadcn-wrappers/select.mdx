```tsx title="ui/select.tsx"
type itemProps = {
  option: allowedPrimitiveT | optionT
  className?: string
}

function Item({ option, className }: itemProps) {
  const isPrimitive = isAllowedPrimitive(option)
  const val = isPrimitive ? option : option.value

  if (val === "---") return <SelectSeparator className={className} />

  return (
    <SelectItem
      value={`${val}`}
      className={cn(className, !isPrimitive && option.className)}
    >
      {isPrimitive ? `${option}` : option.label}
    </SelectItem>
  )
}

type props = {
  options: optionsT
  placeholder?: string
  triggerCls?: string
  contentCls?: string
  groupCls?: string
  itemCls?: string
}
function SelectWrapper({
  options, placeholder,
  triggerCls, contentCls, groupCls, itemCls,
  ...props
}: React.ComponentProps<typeof SelectPrimitive.Root> & props) {
  return (
    <Select {...props}>
      <SelectTrigger className={triggerCls}>
        <SelectValue placeholder={placeholder} />
      </SelectTrigger>

      <SelectContent className={contentCls}>
        {
          options.map((option, i) => {
            if (typeof option === "object" && "group" in option) {
              return (
                <SelectGroup key={option.group} className={cn(groupCls, option.className)}>
                  <SelectLabel className="pb-0.5">{option.group}</SelectLabel>

                  {option.options.map((grOpts, j) => (
                    <Item
                      key={`${option.group}-item-${j}`}
                      option={grOpts}
                      className={cn("pl-4", itemCls)}
                    />
                  ))}
                </SelectGroup>
              )
            }

            const key = isAllowedPrimitive(option)
              ? `${option}-${i}`
              : `${option.value}-${i}`

            return (
              <Item
                key={key}
                option={option}
                className={itemCls}
              />
            )
          })
        }
      </SelectContent>
    </Select>
  )
}
```

Add `isAllowedPrimitive` from `@/lib/utils` and add `SelectWrapper` at the export block.

## Usage

### Basic
```tsx
import { SelectWrapper } from "@/components/ui/select"

export function BasicSelect() {
  return (
    <SelectWrapper
      options={["Option 1", "Option 2", "Option 3"]}
      placeholder="Select an option"
    />
  )
}
```

### Controlled
```tsx
import { useState } from "react"
import { SelectWrapper } from "@/components/ui/select"
import { parseAllowedPrimitive } from "@/lib/utils"

export function ControlledSelect1() {
  const [value, setValue] = useState("apple")

  return (
    <SelectWrapper
      value={value}
      onValueChange={setValue}
      options={["apple", "banana", "orange"]}
    />
  )
}

export function ControlledSelect2() {
  const [value, setValue] = useState<allowedPrimitiveT>("apple")

  return (
    <SelectWrapper
      value={`${value}`}
      onValueChange={(v) => setValue(parseAllowedPrimitive(v))}
      options={["apple", 10, false]}
    />
  )
}
```

### Custom Styling
```tsx
export function ComplexSelect() {
  return (
    <SelectWrapper
      options={["Small", "Medium", "Large", {}]}
      placeholder="Select size"
      triggerCls="w-[200px] bg-slate-50"
      contentCls="min-w-[200px]"
      itemCls="cursor-pointer hover:bg-slate-100"
      groupCls="py-2"
    />
  )
}
```
