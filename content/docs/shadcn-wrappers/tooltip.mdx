Copy and paste the following code into shadcn `tooltip` component.

```tsx title="ui/tooltip.tsx"
type TooltipWrapperProps = {
  trigger: React.ReactNode
  triggerCls?: string
  description: React.ReactNode
  descriptionCls?: string
  contentProps?: Omit<React.ComponentProps<typeof TooltipPrimitive.Content>, "className">
} & Omit<React.ComponentProps<typeof TooltipPrimitive.Provider>, "children">

function TooltipWrapper({
  trigger,
  triggerCls,
  description,
  descriptionCls,
  contentProps,
}: TooltipWrapperProps) {
  return (
    <Tooltip>
      <TooltipTrigger
        className={cn(triggerCls)}
        asChild={typeof trigger !== "string"}
      >
        {trigger}
      </TooltipTrigger>

      <TooltipContent {...contentProps} className={cn(descriptionCls)}>
        {description}
      </TooltipContent>
    </Tooltip>
  )
}
```

Updated `TooltipWrapper` in the export block.

```tsx
export {
  Tooltip,
  TooltipTrigger,
  TooltipContent,
  TooltipProvider,
// [!code focus] [!code ++:1]
  TooltipWrapper,
}
```

## Usage

### Basic
```tsx
import { TooltipWrapper } from "@/components/ui/tooltip";

export function Basic() {
  return (
     <TooltipWrapper
      trigger="Hover"
      description="Some tooltip description"
    />
  )
}
```
