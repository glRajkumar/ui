## Usage

### Basic
```tsx
import { type menubarOptionsT, MenubarCheckboxWrapper, MenubarRadioWrapper, MenubarWrapper } from "@/components/ui/dropdown"

export function Basic() {
    const opts: menubarOptionsT = [
    {
      key: "1",
      trigger: "Option 1",
      options: dropdownOptions
    },
    {
      key: "2",
      trigger: "Option 2",
      options: dropdownOptions,
    }
  ]

  return (
    <>
      <MenubarWrapper
        options={opts}
        contentProps={{ align: "end" }}
      />

      <MenubarCheckboxWrapper 
        options={opts}
      />


      <MenubarRadioWrapper 
        options={opts}
      />
    </>
  )
}
```

### Controlled
```tsx
import { useState } from "react"
import { type menubarOptionsT, MenubarCheckboxWrapper, MenubarRadioWrapper, MenubarWrapper } from "@/components/ui/dropdown"
import { Button } from "@/components/ui/button"

export function Controlled() {
  const [checked, setChecked] = useState<allowedPrimitiveT[]>([])
  const [val, setVal] = useState<allowedPrimitiveT>(true)

  const opts: menubarOptionsT = [
    {
      key: "1",
      trigger: "Option 1",
      options: dropdownOptions
    },
    {
      key: "2",
      trigger: "Option 2",
      options: dropdownOptions,
    }
  ]

  return (
    <>
      <MenubarWrapper
        options={opts}
        onSelect={v => console.log(v)}
      />

      <MenubarCheckboxWrapper
        checked={checked}
        options={opts}
        onCheckedChange={(value, isChecked) => {
          setChecked((prev) =>
            isChecked
              ? [...prev, value]
              : prev.filter((x) => x !== value)
          )
        }}
      />

      <MenubarRadioWrapper
        value={val}
        options={opts}
        onValueChange={setVal}
      />
    </>
  )
}
```

### Nested Complex Data
```tsx
const dropdownOptions = [
  { label: "New File", value: "new", shortcut: "Ctrl+N" },
  "Save",
  12,
  { label: <><Banana /> Banana</>, value: "banana" },

  {
    group: "Settings",
    options: [
      { label: "Appearance", value: "appearance" },
      22,
      true
    ],
  },

  {
    submenu: "More",
    options: [
      { label: <><Apple /> Apple</>, value: "apple" },
      {
        group: "Tools",
        options: [
          { label: "Formatter", value: "formatter" },
          false,
        ],
      },
    ],
  },
]

  const opts: menubarOptionsT = [
    {
      key: "1",
      trigger: "Option 1",
      options: dropdownOptions
    }
  ]

  return (
    <MenubarWrapper
      options={opts}
    />
  )
}
```

  /**
   * Array of menubar menu configurations
   * Each menu is a separate dropdown in the menubar
   * 
   * \`\`\`tsx
   * options={[
   *   {
   *     key: "file",
   *     trigger: "File",
   *     options: ["New", "Open", "Save"]
   *   },
   *   {
   *     key: "edit",
   *     trigger: "Edit",
   *     options: ["Cut", "Copy", "Paste"]
   *   }
   * ]}
   * \`\`\`
   */
  options: MenubarOption[];

  /**
   * Additional CSS classes applied to all menu items
   * Can be overridden per menu or per item
   */
  itemCls?: string;

  /**
   * Additional CSS classes applied to all menu groups
   * Can be overridden per menu or per group
   */
  groupCls?: string;

  /**
   * Additional CSS classes applied to all group labels
   * Can be overridden per menu or per group
   */
  groupLabelCls?: string;

  /**
   * Default props passed to all MenubarContent components
   * Can be overridden per menu via \`contentProps\`
   * 
   * \`\`\`tsx
   * contentProps={{
   *   align: "start",
   *   sideOffset: 5
   * }}
   * \`\`\`
   */
  contentProps?: React.ComponentProps<typeof MenubarContent>;

  /**
   * Global callback fired when any menu item is selected
   * Can be overridden per menu via individual \`onSelect\`
   * 
   * @param value - The value of the selected item
   */
  onSelect?: (value: string | number | boolean) => void;

  /**
   * ...otherProps: All properties from Radix UI Menubar Root can be passed directly.
   * Excludes: children, asChild, value
   */
  otherProps: MenubarPrimitive.Root
}`}
